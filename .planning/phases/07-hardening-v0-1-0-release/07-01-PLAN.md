**File 2 — .github/workflows/publish.yml** (TestPyPI → PyPI → GitHub Release on tag push):

```yaml
name: Publish

on:
  push:
    tags: ["v*"]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - run: pip install build && python -m build
      - uses: actions/upload-artifact@v4
        with:
          name: python-package-distributions
          path: dist/

  test-publish:
    needs: [build]
    runs-on: ubuntu-latest
    environment:
      name: testpypi
      url: https://test.pypi.org/p/agent-factory
    permissions:
      id-token: write
    steps:
      - uses: actions/download-artifact@v4
        with:
          name: python-package-distributions
          path: dist/
      - uses: pypa/gh-action-pypi-publish@release/v1
        with:
          repository-url: https://test.pypi.org/legacy/

  publish:
    needs: [build, test-publish]
    runs-on: ubuntu-latest
    environment:
      name: pypi
      url: https://pypi.org/p/agent-factory
    permissions:
      id-token: write
    steps:
      - uses: actions/download-artifact@v4
        with:
          name: python-package-distributions
          path: dist/
      - uses: pypa/gh-action-pypi-publish@release/v1

  github-release:
    needs: [build, publish]
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: actions/download-artifact@v4
        with:
          name: python-package-distributions
          path: dist/
      - uses: softprops/action-gh-release@v2
        with:
          files: |
            dist/*.whl
            dist/*.tar.gz
```

**Key constraints (from RESEARCH.md pitfalls):**
- `id-token: write` is per-job, NOT at workflow level — security risk otherwise
- `fetch-depth: 0` is REQUIRED for truffleHog — without it, only 1 commit is scanned
- Triggered by `tags: ["v*"]` only — prevents TestPyPI HTTP 400 on re-runs (same version collision)
- Environment names `testpypi` / `pypi` must match what is registered on PyPI Trusted Publisher settings
- `docker compose` (space, v2) already established in smoke-test.yml — publish.yml uses `python -m build` not docker
