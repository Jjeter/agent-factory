```python
def cluster_cli() -> None:

def db() -> None:

@db.command(name="up")

def db_up(db_path: str) -> None:
    """Initialize the cluster database (idempotent)."""
    from runtime.database import DatabaseManager
    asyncio.run(_db_up(Path(db_path)))

async def _db_up(path: Path) -> None:
    mgr = DatabaseManager(path)
```

Key rules:
- Import `DatabaseManager` INSIDE the command body (lazy import) — not at module level — to avoid circular import risk and to keep CLI startup fast
- Use `asyncio.run()` once per command, wrapping a single async coroutine — never nested
- Error handling: let exceptions propagate naturally (Click will display them); no try/except that swallows errors in Phase 1
- Module docstring: `"""Click CLI entry points for cluster and agent-factory commands."""`

```python
@cluster_cli.group(name="db")
def db_group() -> None:
    """Database lifecycle commands."""

@db_group.command(name="up")
@click.option(
    "--db-path",
    default="cluster.db",
    envvar="CLUSTER_DB_PATH",
    show_default=True,
    help="Path to the SQLite database file.",
)
def db_up(db_path: str) -> None:
    """Initialize the cluster database schema (idempotent)."""
    asyncio.run(_do_up(Path(db_path)))
    click.echo(f"Database initialized: {db_path}")

async def _do_up(path: Path) -> None:
    from runtime.database import DatabaseManager
    mgr = DatabaseManager(path)
    await mgr.up()
```

```python
import pytest
from pydantic import ValidationError
from runtime.models import (
    Goal, Task, TaskComment, TaskReview,
    AgentStatus, Document, ActivityLog, TaskStatus
)

def test_goal_model_roundtrip():
    g = Goal(title="Test Goal", description="A test")
    assert g.id  # auto-generated UUID
    assert g.created_at  # auto-generated ISO timestamp
    g2 = Goal.model_validate(g.model_dump())
    assert g2.id == g.id
    assert g2.title == g.title
    assert g2.status == "active"

def test_task_rejects_invalid_status():
    with pytest.raises(ValidationError):
        Task(goal_id="g1", title="T", description="D", status="invalid_status")

def test_task_status_raw_string():
    t = Task(goal_id="g1", title="T", description="D", status="todo")
    assert t.status == "todo"  # use_enum_values=True serializes to string

def test_all_models_construct():
    g = Goal(title="G", description="D")
    t = Task(goal_id=g.id, title="T", description="D")
    tc = TaskComment(task_id=t.id, agent_id="a1", comment_type="feedback", content="c")
    tr = TaskReview(task_id=t.id, reviewer_id="r1")
    a = AgentStatus(id="boss", agent_role="boss")
    d = Document(title="doc", content="text", created_by="a1")
    al = ActivityLog(agent_id="a1", action="task_claimed")
    # All constructed without error
    assert g.id and t.id and tc.id and tr.id and a.id and d.id and al.id

def test_task_status_no_rejected_value():
    values = [s.value for s in TaskStatus]
    assert "rejected" not in values
    assert len(values) == 5

def test_agent_status_defaults():
    a = AgentStatus(id="agent-1", agent_role="researcher")
    assert a.status == "idle"
    assert a.last_heartbeat is None
    assert a.current_task is None
```
