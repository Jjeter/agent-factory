From RESEARCH.md Pattern 1 (DatabaseManager skeleton):
```python
import aiosqlite
from pathlib import Path

STARTUP_PRAGMAS = [
    "PRAGMA journal_mode = WAL",
    "PRAGMA synchronous = NORMAL",
    "PRAGMA foreign_keys = ON",
    "PRAGMA busy_timeout = 5000",
]

class DatabaseManager:
    def __init__(self, db_path: Path) -> None:
        self._db_path = db_path

    async def _apply_pragmas(self, db: aiosqlite.Connection) -> None:
        for pragma in STARTUP_PRAGMAS:
            await db.execute(pragma)

    async def open_write(self) -> aiosqlite.Connection:
        db = await aiosqlite.connect(self._db_path)
        db.row_factory = aiosqlite.Row
        await self._apply_pragmas(db)
        return db

    async def open_read(self) -> aiosqlite.Connection:
        db = await aiosqlite.connect(self._db_path)
        db.row_factory = aiosqlite.Row
        await self._apply_pragmas(db)
        return db

    async def init_schema(self, db: aiosqlite.Connection) -> None:
        schema = (Path(__file__).parent / "schema.sql").read_text(encoding="utf-8")
        await db.executescript(schema)
        # executescript implicitly commits; explicit commit for clarity
        await db.commit()

    async def up(self) -> None:
        """Initialize schema (idempotent). Used by CLI 'cluster db up'."""
        db = await self.open_write()
        try:
            await self.init_schema(db)
        finally:
            await db.close()

    async def reset(self) -> None:
        """Drop all tables and recreate from schema. DESTRUCTIVE."""
        db = await self.open_write()
        try:
            # Drop in reverse dependency order
            await db.executescript("""
                DROP TABLE IF EXISTS activity_log;
                DROP TABLE IF EXISTS documents;
                DROP TABLE IF EXISTS task_reviews;
                DROP TABLE IF EXISTS task_comments;
                DROP TABLE IF EXISTS agent_status;
                DROP TABLE IF EXISTS tasks;
                DROP TABLE IF EXISTS goals;
            """)
            await self.init_schema(db)
        finally:
            await db.close()
```

Pragma persistence rules (from RESEARCH.md Pitfall 2):
- journal_mode=WAL IS persistent (stored in DB file)
- foreign_keys, busy_timeout, synchronous are NOT persistent — must be set per-connection
- Set all 4 on every connection opened (idempotent, zero cost)

aiosqlite 0.22.0+ breaking change (from RESEARCH.md Pitfall 1):
- No threading.Thread inheritance — must always await db.close() or use async with
- For tests: use try/finally to guarantee close()

executescript behavior (from RESEARCH.md Pitfall 3):
- Issues implicit COMMIT before executing — safe for init_schema (no prior transaction)
- Do NOT call executescript mid-transaction

Row → model pattern (from RESEARCH.md open question 3):
- `Model.model_validate(dict(row))` where row is aiosqlite.Row
- Never pass aiosqlite.Row directly to Pydantic — convert to dict first

Key implementation rules:

1. Use individual `await db.execute(pragma)` calls for each pragma — NOT `await db.executescript(all_pragmas)`. This avoids the implicit COMMIT of executescript during normal connection setup. Only use executescript for schema initialization.

2. `init_schema()` reads `(Path(__file__).parent / "schema.sql").read_text(encoding="utf-8")` — this resolves correctly when runtime/ is the package directory alongside schema.sql.

3. `reset()` drop order must respect FK dependencies: activity_log, documents, task_reviews, task_comments, agent_status, tasks, goals (reverse of creation order). Drop with `DROP TABLE IF EXISTS` to be safe.

4. Do NOT implement any insert/update/delete methods in DatabaseManager — it is a connection factory only. Higher-level code (agent classes in Phase 2+) will use the raw connection to execute SQL.

5. Add a module docstring: `"""SQLite database connection manager with WAL mode and per-connection pragma setup."""`

6. Export only `DatabaseManager` from this module.
