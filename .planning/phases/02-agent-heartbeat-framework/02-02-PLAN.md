
Interruptible sleep pattern (stop event wakes loop immediately):
```python
try:
    await asyncio.wait_for(self._stop_event.wait(), timeout=sleep_secs)
except asyncio.TimeoutError:
    pass  # Normal: timeout means interval elapsed, continue loop
# CancelledError propagates through wait_for naturally — do NOT catch it
```

CancelledError inheritance (Python 3.8+):
- asyncio.CancelledError inherits from BaseException (NOT Exception)
- `except Exception` does NOT catch CancelledError — safe in _tick()
- NEVER use `except BaseException` or bare `except:` — these would swallow CancelledError

**`async def start(self) -> None`:**
Entry point. Stagger first, then loop.
```python
async def start(self) -> None:
    if self._config.stagger_offset_seconds > 0:
        await asyncio.sleep(self._config.stagger_offset_seconds)
    try:
        while not self._stop_event.is_set():
            await self._tick()
            jitter = random.uniform(-30, 30)
            sleep_secs = max(0.0, self._config.interval_seconds + jitter)
            try:
                await asyncio.wait_for(self._stop_event.wait(), timeout=sleep_secs)
            except asyncio.TimeoutError:
                pass  # Normal: interval elapsed, continue loop
    finally:
        await self._on_shutdown()
```
CRITICAL: No `except CancelledError` anywhere. `CancelledError` propagates through
`asyncio.wait_for` and the while loop into the `finally` block, then re-raises. This is
the correct behavior — external cancellation triggers cleanup then propagates.

**`async def _tick(self) -> None`:**
One heartbeat cycle. Status: working at start, idle at end.
```python
async def _tick(self) -> None:
    await self._set_db_status(AgentStatusEnum.WORKING)
    try:
        await self.do_peer_reviews()
        await self.do_own_tasks()
        await self._log_heartbeat()
        await self._set_db_status(AgentStatusEnum.IDLE)
    except Exception:
        logger.exception(
            "Unhandled error in tick for agent %s", self._config.agent_id
        )
        await self._set_db_status(AgentStatusEnum.ERROR)
    await self._write_state_file()
```
Note: `_write_state_file()` is called unconditionally AFTER the try/except — it always
runs whether the tick succeeded or errored. The state file reflects "what actually happened"
after DB writes complete.

```python
async def _set_db_status(self, status: AgentStatusEnum) -> None:
    db = await self._db.open_write()
    try:
        await db.execute(
            """INSERT INTO agent_status (id, agent_role, status, last_heartbeat, current_task)
               VALUES (?, ?, ?, ?, ?)
               ON CONFLICT(id) DO UPDATE SET
                   status=excluded.status,
                   last_heartbeat=excluded.last_heartbeat,
                   current_task=excluded.current_task""",
            (
                self._config.agent_id,
                self._config.agent_role,
                status.value,
                _now_iso(),
                None,
            ),
        )
        await db.commit()
    finally:
        await db.close()
```
Use `status.value` because `AgentStatusEnum` has `use_enum_values=True` in models but
the raw `.value` is safer for explicit DB writes.

Atomic write via tmp + Path.replace(). Creates STATE_DIR if needed.

```python
async def _write_state_file(self) -> None:
    STATE_DIR.mkdir(parents=True, exist_ok=True)
    state = {"last_heartbeat": _now_iso(), "current_task_id": None}
    tmp = self._state_path.with_suffix(".tmp")
```

**`def _load_state(self) -> dict`:**
Load local state file. Handles missing/corrupt files gracefully.
```python
def _load_state(self) -> dict:
    try:
        return json.loads(self._state_path.read_text(encoding="utf-8"))
    except (FileNotFoundError, json.JSONDecodeError):
        logger.warning(
            "State file missing or corrupt for agent %s — treating as fresh start",
            self._config.agent_id,
```
